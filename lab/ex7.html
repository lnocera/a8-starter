1 - Getting GeoJSON

Download GeoJSON using the Export menu from the Los Angeles GeoHub Community Boundaries (CSA) and save it in data/Community Boundaries (CSA).geojson.

Complete the Answer where indicated by üöß.

Answer
d3.json('data/Community Boundaries (CSA).geojson').then(json => {
var svg = d3.select('#svg1'),
  width = +svg.attr('width'),
  height = +svg.attr('height');

var projection = d3.geoMercator()  //üöß  explain
  .fitSize([width, height], json);

path = d3.geoPath()  //üöß  explain
  .projection(projection);  //üöß  explain

svg.selectAll('path')
  .data(json.features)  //üöß  explain
  .enter()
  .append('path')
  .attr('fill', 'white')
  .attr('stroke', 'black')
  .attr('d', path);
});

d3.json('data/Community Boundaries (CSA).geojson').then(json => {

var svg = d3.select('#svg2'),

  height = +svg.attr('height');

  .fitSize([width, height], json);

path = d3.geoPath()
  .projection(projection);

svg.append('g')
  .attr('class', 'cities')
  .selectAll('path')
  .data(json.features)
  .enter()
  .append('path')
  .attr('d', path)
  .on('mouseenter', function (d) {
    svg.select('.selected')
      .classed('selected', false);
    d3.select(this)
      .classed('selected', true);
    d3.select('#info2')
      .text('properties = ' + JSON.stringify(this.__data__.properties, null, 2));  //üöß explain
  });
});

  json.features = json.features.filter(d => {

            //d.properties.objectid !== "" &&  //üöß  finish to filter out the islands here
