4 - drawing a feature collection

Complete the Answer where indicated by ğŸš§.

Answer
d3.json('data/us.json').then(json => {  //ğŸš§  explain

  var svg = d3.select('#svg4'),
    width = +svg.attr('width'),
    height = +svg.attr('height');

  var projection = d3.geoAlbersUsa()
    .fitSize([width, height], json);

  var path = d3.geoPath().projection(projection);  //ğŸš§  explain

  var graticule = d3.geoGraticule()  //ğŸš§  explain
    .step([10, 10]);

  svg.append('path')
    .datum(graticule)  //ğŸš§  explain
    .attr('class', 'graticule')
    .attr('d', path);

  svg.append('path')
    .attr('fill', 'rgb(255,190,232)')
    .attr('stroke', 'black')
    .attr('d', path(json.features[4]));  //generate geographic path

  svg.selectAll('.states')
    .data(json.features)  //ğŸš§  explain
    .enter()
    .append('path')
    .attr('fill', d => 'white')  //ğŸš§  explain
    //.attr('fill', d => d.id == '06' ? '#FFCC00' : 'white')  //ğŸš§  try and explain
    .attr('class', 'states')
    .attr('d', path);

  //ğŸš§  explain
  var marker = projection([-118.289152, 34.019350]);
  svg.append('circle')
    .attr('cx', marker[0])
    .attr('cy', marker[1])
    .attr('r', 5);
});
